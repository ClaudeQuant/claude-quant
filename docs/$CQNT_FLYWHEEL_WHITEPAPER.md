## Core Mechanics


### Step 1: Token Purchase

Users buy your utility token on Pump.fun
This creates initial liquidity and establishes market value
Token holders become stakeholders in the fund's performance

### Step 2: Creator Fee Collection

Pump.fun generates creator fees from trading volume
These fees accumulate as dedicated capital
Transparent tracking of all fee collection (following Claude Quant's transparency principle)

### Step 3: Investment in Claude Quant Fund

Accumulated creator fees are systematically invested in $CQNT
Investment follows a rules-based allocation schedule (e.g., weekly or monthly)
Position sizes adjust based on market conditions and available capital
All investments publicly documented with wallet addresses

### Step 4: Profit Generation

Claude Quant Fund generates returns through their systematic futures trading
Expected returns based on their backtested performance (~20% monthly targets)
Monthly performance verified through their planned CPA audits
Your token holders benefit from professionally managed trading strategy

### Step 5: Monthly Buyback Execution

Profits from CQNT positions are realized on predetermined schedule
Fixed percentage (e.g., 70-80%) allocated to token buybacks
Remaining percentage reinvested or reserved for operational costs
Buybacks executed systematically to avoid market manipulation concerns

### Step 6: Token Price Appreciation

Buybacks reduce circulating supply
Decreased supply + sustained demand = price appreciation
Rising price attracts new buyers and increases trading volume

### Step 7: Increased Creator Fees (Loop Restart)

Higher token price and more trading volume generate larger creator fees
Larger fees enable bigger CQNT investments
Bigger investments produce greater absolute profits
Greater profits fund larger buybacks
The flywheel accelerates

contract TreasuryManager {
uint256 public constant STARTING_CAPITAL = 100_000 * 10**6; // $100k in USDC
uint256 public launchTimestamp;

struct Phase {
uint256 startMonth;
uint256 endMonth;
uint256 buybackBps;    // Basis points (10000 = 100%)
uint256 reinvestBps;
string name;
}

Phase[4] public phases = [
Phase(1, 3, 10000, 0, "Ignition"),      // 100% buyback
Phase(4, 6, 5000, 5000, "Momentum"),    // 50/50 split
Phase(7, 9, 2500, 7500, "Transition"),  // 25/75 split
Phase(10, type(uint256).max, 1000, 9000, "Exponential") // 10/90 perpetual
];

function getCurrentPhase() public view returns (Phase memory) {
uint256 monthsSinceLaunch = (block.timestamp - launchTimestamp) / 30 days;

for (uint i = 0; i < phases.length; i++) {
if (monthsSinceLaunch >= phases[i].startMonth &&
monthsSinceLaunch <= phases[i].endMonth) {
return phases[i];
}
}

return phases[3]; // Default to perpetual phase
}

function calculateAllocation(uint256 profitAmount) public view returns (
uint256 buybackAmount,
uint256 reinvestAmount
) {
Phase memory currentPhase = getCurrentPhase();

buybackAmount = (profitAmount * currentPhase.buybackBps) / 10000;
reinvestAmount = (profitAmount * currentPhase.reinvestBps) / 10000;
}
}

## Price Impact Protection

async function executeSafeBuyback(targetAmount) {
const pool = await getPoolData();

// Check liquidity depth
const availableLiquidity = pool.reserves.token;
const maxSafeAmount = availableLiquidity * 0.02; // Max 2% of pool

if (targetAmount > maxSafeAmount) {
// Split into multiple transactions
const numTrades = Math.ceil(targetAmount / maxSafeAmount);
const tradeAmount = targetAmount / numTrades;

for (let i = 0; i < numTrades; i++) {
await executeSingleBuyback(tradeAmount);
await sleep(300000); // 5 min delay between trades
}
} else {
await executeSingleBuyback(targetAmount);
}
}

async function executeSingleBuyback(amount) {
const slippageTolerance = 0.03; // 3% max
const deadline = Date.now() + 600000; // 10 min

const quote = await getSwapQuote(amount);
const minTokensOut = quote.tokensOut * (1 - slippageTolerance);

return await swapAndBurn({
solIn: amount,
minTokensOut: minTokensOut,
deadline: deadline
});
}
